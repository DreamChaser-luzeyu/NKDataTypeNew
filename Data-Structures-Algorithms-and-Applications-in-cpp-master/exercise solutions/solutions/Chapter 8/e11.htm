The code is very similar to the case when the binary tree is represented using pointers. The code is given below and in the file btraver.cpp. 

template <class T>

void LevelOrder(T a[], int last)

{// Level-order traversal of a binary tree.

   LinkedQueue<int> Q;

   if (!last) return; // empty tree

   int i = 1; // start at root

   while (true) {

      Visit(a, i);  



      // put children on queue

      if (2*i <= last && a[2*i])

         Q.Add(2*i);       // add left child to queue

      if (2*i+1 <= last && a[2*i+1])

         Q.Add(2*i+1);  // add right child to queue



      // get next node to visit

      try {Q.Delete(i);}

      catch (OutOfBounds) {return;}

      }

 }
