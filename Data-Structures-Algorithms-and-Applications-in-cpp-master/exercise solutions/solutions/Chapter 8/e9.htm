template <class T>

void InOrder(T a[], int last, int i)

{// Inorder traversal of binary tree with root a[i].

   if (i <= last && a[i]) {

      InOrder(a, last, 2*i);    // do left subtree

      Visit(a, i);              // visit tree root

      InOrder(a, last, 2*i+1);  // do right subtree

      }

}
